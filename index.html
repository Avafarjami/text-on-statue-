function onSelect() { if (isPlaced) { const raycaster = new THREE.Raycaster(); const tempMatrix = new THREE.Matrix4(); tempMatrix.identity().extractRotation(camera.matrixWorld); raycaster.ray.origin.setFromMatrixPosition(camera.matrixWorld); raycaster.ray.direction.set(0, 0, -1).applyMatrix4(tempMatrix); const intersects = raycaster.intersectObject(referenceCube, true); if (intersects.length > 0) { const clickedMesh = intersects[0].object; // سطحی که در بلندر جدا کردی console.log("سطح انتخاب شده:", clickedMesh.name); const msg = prompt("متن یادگاری برای این بخش:"); if (msg) { // ارسال مرکز سطح و جهت نرمال آن برای چسبیدن دقیق addLabelToSurface(msg, clickedMesh, intersects[0].point, intersects[0].face.normal); } } } else if (reticle.visible) { // ... (همان منطق قبلی برای فیکس کردن مدل) referenceCube.position.setFromMatrixPosition(reticle.matrix); referenceCube.quaternion.setFromRotationMatrix(reticle.matrix); referenceCube.visible = true; isPlaced = true; } } function addLabelToSurface(text, mesh, point, normal) { const canvas = document.createElement('canvas'); const ctx = canvas.getContext('2d'); // بزرگتر کردن رزولوشن برای وضوح بیشتر در ابعاد بزرگ canvas.width = 1024; canvas.height = 256; // پس‌زمینه نیمه‌شفاف برای خوانایی (اختیاری) ctx.fillStyle = 'rgba(0, 0, 0, 0.4)'; ctx.fillRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = 'white'; ctx.font = 'bold 80px Arial'; // فونت بسیار بزرگتر ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.fillText(text, 512, 128); const texture = new THREE.CanvasTexture(canvas); const material = new THREE.MeshBasicMaterial({ map: texture, transparent: true, side: THREE.DoubleSide, depthTest: true }); // اندازه فیزیکی برچسب (اینجا می‌توانید عرض را مثلا به 0.8 متر برسانید) const geometry = new THREE.PlaneGeometry(0.6, 0.15); const label = new THREE.Mesh(geometry, material); // قرار دادن دقیق در محل کلیک label.position.copy(point); // چسباندن به سطح: جهت برچسب را با جهت سطح (Normal) یکی می‌کنیم const lookAtPoint = new THREE.Vector3().addVectors(point, normal); label.lookAt(lookAtPoint); // یک فاصله بسیار کم (Offset) برای جلوگیری از تداخل با سطح مجسمه label.position.add(normal.clone().multiplyScalar(0.02)); scene.add(label); }
